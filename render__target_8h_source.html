38 /// @class RenderTarget
39 /// @brief Abstracts a surface that can be rendered to.
40 ///
41 /// Calling SetAsRenderTarget() will cause all subsequent draw calls to be
42 /// drawn onto the RenderTarget instead of to the screen buffer.
43 class RenderTarget {
44  public:
45  RenderTarget() : initialized_(false) {}
46  /// @brief Initialize a render target of the provided dimensions.
47  ///
48  /// Defaults the format to GL_UNSIGNED_BYTE, using a depth buffer.
49  ///
50  /// @param dimensions The dimensions of the render target.
51  void Initialize(mathfu::vec2i dimensions);
52  /// @brief Initialize a render target of the provided dimensions.
53  ///
54  /// @param dimensions The dimensions of the render target.
55  /// @param format The OpenGL format of the generated texture.
56  /// @param useDepthBuffer Create a depth buffer used by the render target.
57  void Initialize(mathfu::vec2i dimensions, GLenum format, bool useDepthBuffer);
58 
59  /// @brief Deletes the associated opengl resources associated with the
60  /// RenderTarget.
61  void Delete();
62 
63  /// @brief Sets the RenderTarget as the active render target.
64  ///
65  /// All subsequent openGL draw calls will render to this RenderTarget
66  /// instead of wherever they were going before.
67  void SetAsRenderTarget() const;
68 
69  /// @brief Binds the texture associated with this rendertarget as the active
70  /// texture.
71  ///
72  /// Primarily useful when rendering the RenderTarget's texture as part of a
73  /// mesh. Throws an assert if the RenderTarget does not have a texture.
74  ///
75  /// @param texture_number The index of the texture to make active.
76  void BindAsTexture(int texture_number) const;
77 
78  /// @brief Checks if this rendertarget refer to an off-screen texture.
79  ///
80  /// This is important because rendertargets that aren't texture-based will
81  /// assert if you try to bind them as texture or access their textureId.
82  ///
83  /// @return Returns true if this rendertarget refers to an off-screen texture,
84  /// and false if it refers to the screen itself.
85  inline bool IsTexture() const { return framebuffer_id_ != 0; }
86 
87  /// @brief Gets the TextureId associated with the RenderTarget, assuming that
88  /// it is texture-based.
89  ///
90  /// Throws an assert if you try to call GetTextureId on a RenderTarget that
91  /// doesn't have a texture backing it, such as the screen's display buffer.
92  ///
93  /// @return Returns the TextureId associated with the RenderTarget.
94  inline unsigned int GetTextureId() const {
95  assert(IsTexture());
96  return rendered_texture_id_;
97  }
98 
99  /// @brief Checks if the RenderTarget has been initialized.
100  ///
101  /// Returns true if this RenderTarget has been initialized and is ready to
102  /// use. Returns false if has not yet been initialized, failed initalization,
103  /// or has been deleted. Trying to use an uninitialized RenderTarget will
104  /// generally cause errors or asserts.
105  ///
106  /// @brief Returns true if it has been initialized, false otherwise.
107  bool initialized() const { return initialized_; }
108 
109  /// @brief Gets the RenderTarget that corresponds to the screen.
110  ///
111  /// @param renderer The renderer object to use.
112  /// @return Returns the RenderTarget that corresponds to the screen.
113  static RenderTarget ScreenRenderTarget(Renderer& renderer);
114 
115  private:
116  mathfu::vec2i dimensions_;
117  GLuint framebuffer_id_;
118  GLuint rendered_texture_id_;
119  GLuint depth_buffer_id_;
120  bool initialized_;
121 };
